export const PatchVersion = () => {
  if (typeof document !== "undefined") {
    requestAnimationFrame(() => {
      document.querySelectorAll("[data-versioned]").forEach(a => {
        a.href = a.href.replace("VERSION_PLACEHOLDER", "stable");
      });
    });
  }
  return null;
};
